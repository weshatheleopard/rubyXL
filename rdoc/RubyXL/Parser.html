def self.parse(src_file_path)
  begin
    ::Zip::File.open(src_file_path) { |zip_file|
      root = RubyXL::WorkbookRoot.parse_zip_file(zip_file)
      root.source_file_path = src_file_path
      root.workbook
    }
  rescue ::Zip::Error => e
    raise e, "XLSX file format error: #{e}", e.backtrace
  end
end
