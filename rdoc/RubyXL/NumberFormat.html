def is_date_format?
  #             v-------- Toss all the escaped chars -------v v--- and see if any date-related remained
  !!(format_code.gsub(/(\"[^\"]*\"|\[[^\]]*\]|[\\_*].)/i, '') =~ /[dmyhs]/i)
end
