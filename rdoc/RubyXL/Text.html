def before_write_xml
  preserve_whitespace
  self.value.gsub!(INVALID_XML10_CHARS) { |c| "_x%04x_" % c.ord }
  true
end

def to_s
  value.to_s.gsub(ESCAPED_UNICODE) { |m| $1.hex.chr(::Encoding::UTF_8) }
end
