def self.chars2raw(width_in_chars)
  ((width_in_chars + (5.0 / RubyXL::Font::MAX_DIGIT_WIDTH)) * 256).to_i / 256.0
end

def delete_column(col_index)

  self.min -= 1 if min >= col

def insert_column(col_index)
  col = col_index + 1
  self.min += 1 if min >= col
  self.max += 1 if max >= col - 1
end
